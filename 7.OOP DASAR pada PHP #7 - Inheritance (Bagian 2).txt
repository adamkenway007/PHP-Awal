<!-- Inheritance Part 2 = 
  - Menciptakan hierarki antar kelas (Parent & Child)
  - Child Class, mewarisi semua properti dan method dari parent-nya (yang 
    visible)
  - Child Class, memperluas (extends) fungsionalitas dari parent-nya
-->
<?php
 
 class Mobil {
 	public $nama, $merk, $warna,
 		   $kecepatanMaksimal,
 		   $jumlahPenumpang;

 	public function tambahKecepatan() {
 		return "Kecepatan Bertambah!";
 	}
 }

 class MobilSport extends Mobil {
 	public $turbo = false;

 	public function jalankanTurbo() {
 		$this->turbo = true;
 		return "Turbo dijalankan!";
 	}
 }

 $mobil = new MobilSport();
 echo $mobil->tambahKecepatan();
 echo "<br>";
 echo $mobil->jalankanTurbo();
